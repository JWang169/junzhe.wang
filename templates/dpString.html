    def longestRepeatingSubstring(self, S: str) -> int:
            n = len(S) + 1
            dp = [[0] * n for _ in range(n)]
            result = 0
            for i in range(1, n):
                for j in range(i + 1, n):
                    if S[i - 1] == S[j - 1]:
                        dp[i][j] = dp[i - 1][j - 1] + 1 
                        result = max(result, dp[i][j])        
            
            return result
