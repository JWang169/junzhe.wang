1 /*
  2 abstract class fulfills Polymorphism, so is interface
  3 sometimes abstract class is not enough, because one class cannot inherit multiple class 
  4 For example, if we need method Hunt() for class Tiger, but it cannot be added to Mammal class, since Goat do not hunt
  5 Tiger needs to implement Hunt() from somewhere else, which is interface.
  6 
  7 */
  8 
  9 interface Huntable{
 10     public void hunt(Animal a);
 11 }
 12 
 13 
 14 // define the father abstract class
 15 // we only define the name of the methods, not the body
 16 abstract class Animal{
 17     public abstract String getName();
 18     public abstract void move(String destination);
 19     public abstract void drink();
 20 }
 21 
 22 // define a sub abstract class, which inherit Animal
 23 abstract class Mammal extends Animal{
 24     // inherit all methods from Animal;
 25     // you can also add new methods here 
 26     public abstract void feedChildren();
 27 }
 28 
 29 
 30 // define another sub abstract class
 31 abstract class Reptile extends Animal{
 32     // inherit all methods from Animal;
 33     // you can also add new methods here 
 34 }
 35 
 36 
 37 // define a class Tiger. it has to rewrite all the abstract methods from the super class
 38 class Tiger extends Mammal implements Huntable{
 39     private static String name = "Tiger";
 40     public String getName(){
 41         return this.name;
 42     }
 43     public void move(String destination){
 44         System.out.println("A tiger runs to " + destination + ".");
 45     }
 46     public void drink(){
 47         System.out.println("A tiger drinks water");
 48     }
 49     public void feedChildren(){
 50         System.out.println("Tigers breastfeeds.");
 51     }
 52     public void hunt(Animal a){
 53         System.out.println("Tiger caught a " + a.getName() + " and ate it.");
 54     }
 55 }
 56

 58 // define another class goat. it also extends mammal 
 59 class Goat extends Mammal{
 60         private static String name = "Goat";
 61     public String getName(){
 62         return this.name;
 63     }
 64     public void move(String destination){
 65         System.out.println("A goat runs to " + destination + ".");
 66     }
 67     public void drink(){
 68         System.out.println("A goat drinks water");
 69     }
 70     public void feedChildren(){
 71         System.out.println("Goats breastfeeds.");
 72     }
 73 }
 74 
 75 class Snake extends Reptile implements Huntable{
 76     private static String name = "Snake";
 77     public String getName(){
 78         return this.name;
 79     }
 80     public void move(String destination){
 81         System.out.println("A Snake crawls to " + destination + ".");
 82     }
 83     public void drink(){
 84         System.out.println("A snake drinks water. ");
 85     }
 86     public void hunt(Animal a){
 87         System.out.println("A Snake caught a " + a.getName() + " and ate it.");
 88     }
 89     // that's it. because snake extends Reptile, no feedChildren method in Reptile;
 90 }
 91 
 92 // create a farmer class
 93 class Farmer{
 94     public void bringWater(String destination){
 95         System.out.println("A farmer brings water to " + destination + ".");
 96     }
 97     public void bringAnimal(Animal a){
 98         System.out.println("A farmer brings " + a.getName() + " to the feeding room.");
 99     }
100     public void feedWater(Animal a){
101         this.bringWater("feeding room");
102         a.move("feeding room");
103         a.drink();
104     }
105     public void feedAnimal(Animal ht, Animal ct){
106         this.bringAnimal(ct);
107         ht.move("Feeding room. ");
108         Huntable htb = (Huntable) ht;  // an interface will allow all class to implement it. 
109         htb.hunt(ct);
110     }
111 }

123         fm.feedAnimal(tg, gt);
